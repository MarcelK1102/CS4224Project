Let m be an empty dictionary with O_ID as key and as value a pair of item id's
\begin{algorithmic}
   \ForAll{ $o_{c}$ in orders of customer c} 
   \ForAll{items $i$ in $o_{c}$} 
   \ForAll{$o_{c'}$ in orders including item $i$} 
   \State $e\gets m[o_{c'}]$
   \If{$e$ is empty}
   \State \Call{insert}{i,-1} in m
   \Else
   \State $(i_{1},i_{2})\gets e$
   \If{$i$ != $i_{1}$ and $i_{2}$ = $-1$}
      \State $m[o_{c'}]\gets (i_{1},i)$ 			  
      \State \Call{output}{c'}
   \EndIf
   \EndIf
   \EndFor
   \EndFor
   \EndFor
\end{algorithmic}
